algorithm: 滑动窗口

调用：[[C++#String|<string>]] [[C++#哈希表|<unordered_map>]]
```cpp
class Solution {
public:
    int lengthOfLongestSubstring(string s) {
        unordered_map<char, int> hash;
        int n = s.size();
        int p = 0, q = 0;
        int maxlen = 0;
        while (q < n) {
            if (hash.count(s[q])) {
                if (hash[s[q]] >= p) {
                    p = hash[s[q]] + 1;
                    hash[s[q]] = q;
                    q++;
                } else {
                    hash[s[q]] = q;
                    q++;
                    maxlen = max(q-p, maxlen);
                }
            } else {
                hash.insert({s[q], q});
                q++;
                maxlen = max(q-p, maxlen);
            }
        }
        return maxlen;
    }
};
```
