#include <iostream>
#include <string>
#include <unordered_map>
using namespace std;

string s;
int count = -1;
unordered_map<string,int> mp;
int len;
bool used[7];

void dfs(string part) {
    if (mp.count(part)) { return; }
    else {
        mp.insert({part, 0});
        count++;
        for (int i=0; i<len; i++) {
            if (!used[i]) {
                part.push_back(s[i]);
                used[i] = true;
                dfs(part);
                part.pop_back();
                used[i] = false;
            }
        }
        return;
    }
}

int main() {
    cin >> s;
    len = s.size();
    for (int i=0; i<len; i++) { used[i] = false; }

    string tmp;
    dfs(tmp);
    cout << count << endl;

    return 0;
}
