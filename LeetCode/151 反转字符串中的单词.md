class Solution {
public:
    string reverseWords(string s) {
        reverse(s.begin(), s.end());
        while (s[0] == ' ') { s.erase(s.begin()); }
        while (s.back() == ' ') { s.pop_back(); }
        s.push_back(' ');
        int n = s.size();
        int p = 0;
        int q;
        while (p < n-1) {
            if (s[p] == ' ') {
                q = p;
                while (s[q] == ' ') { q++; }
                s.erase(s.begin()+p+1, s.begin()+q);
                n -= q-p-1;
                p++;
            } else {
                q = p;
                while (s[q] != ' ') { q++; }
                reverse(s.begin()+p, s.begin()+q);
                p = q;
            }
        }
        s.pop_back();
        return s;
    }
};
