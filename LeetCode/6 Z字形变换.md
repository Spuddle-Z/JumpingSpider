使用了[[C++#String|<string>]]。
```cpp
class Solution {
public:
    string convert(string s, int numRows) {
        int n = s.size();
        if (numRows == 1) { return s; }
        int m = 2*numRows - 2;
        string res;
        int p = 0;
        while (p < n) {
            res.push_back(s[p]);
            p += m;
        }
        for (int i=1; i<numRows-1; i++) {
            p = i;
            while (true) {
                if (p < n) { res.push_back(s[p]); }
                else { break; }
                if (p+m-i-i < n) { res.push_back(s[p+m-i-i]); }
                else { break; }
                p += m;
            }
        }
        p = numRows - 1;
        while (p < n) {
            res.push_back(s[p]);
            p += m;
        }
        return res;
    }
};
```
