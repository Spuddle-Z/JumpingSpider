class Solution {
public:
    int removeDuplicates(vector<int>& nums) {
        int cnt = 0;
        int i = 0;
        while (i < nums.size()-1) {

            if (nums[i] == nums[i+1]) {
                cnt++;
            } else {
                cnt = 0;
            }

            if (cnt == 2) {
                cnt--;
                nums.erase(nums.begin() + i + 1);
            } else {
                i++;
            }
        }
        return nums.size();
    }
};
