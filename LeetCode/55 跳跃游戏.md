class Solution {
public:
    bool canJump(vector<int>& nums) {
        int tmp;
        int i = 0;
        int M = nums[i];
        while (i <= M) {
            if (M >= nums.size()-1) { return true; }
            tmp = i + nums[i];
            if (tmp > M) { M = tmp; }
            i++;
        }
        return false;
    }
};
