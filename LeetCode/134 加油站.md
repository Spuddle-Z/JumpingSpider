使用了[[C++#Vector|<vector>]]。
> [!note] 
> 可以证明，若从`i`无法到达`j`，则`i` `j`之间的所有点都不可能是起点，跳过这些点可以大大减少计算量。
```cpp
class Solution {
public:
    int canCompleteCircuit(vector<int>& gas, vector<int>& cost) {
        int n = gas.size();
        for (int i=0; i<n; i++) { gas[i] -= cost[i]; }

        int start = 0;
        int step, p, cnt;
        while (start < n) {
            step = 0;
            cnt = 0;
            while (step < n) {
                p = (start + step) % n;
                cnt += gas[p];
                if (cnt < 0) {
                    break;
                }
                step++;
            }
            if (step == n) { return start; }
            if (start < p+1) { start = p+1; }
            else { break; }
        }
        return -1;
    }
};
```
