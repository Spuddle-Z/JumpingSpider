class Solution {
public:
    bool isSubsequence(string s, string t) {
        int p = 0, q = 0;
        int m = s.size(), n = t.size();
        while (p < m) {
            while (true) {
                if (q == n) { return false; }
                else if (t[q] == s[p]) {
                    q++;
                    break;
                }
                else { q++; }
            }
            p++;
        }
        return true;
    }
};
