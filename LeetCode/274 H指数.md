---
algorithm: 分治法
---
```cpp
class Solution {
public:
    int hIndex(vector<int>& citations) {
        int n = citations.size();
        sort(citations.rbegin(), citations.rend());
        if (citations[0] == 0) { return 0; }

        int p = 0;
        int q = n;
        int m;
        while (p < q-1) {
            m = (p+q)/2;
            if (m+1 > citations[m]) {
                q = m;
            } else if (m+1 == citations[m]) {
                return m+1;
            } else {
                p = m;
            }
        }
        return p+1;
    }
};
```
