使用了[[C++#Vector|<vector>]]。
```cpp
class Solution {
public:
    vector<int> productExceptSelf(vector<int>& nums) {
        vector<int> L;
        int R = 1;
        L.push_back(1);
        int n = nums.size();
        for (int i=1; i<n; i++) {
            L.push_back(L[i-1]*nums[i-1]);
        }
        for (int i=n-1; i>=0; i--) {
            L[i] *= R;
            R *= nums[i];
        }
        return L;
    }
};
```
