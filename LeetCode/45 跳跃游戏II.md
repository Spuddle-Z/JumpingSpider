class Solution {
public:
    int jump(vector<int>& nums) {
        if (nums.size() == 1) { return 0; }
        int cnt = 1;
        int tmp;
        int m = 0;
        int M = nums[0];
        while (M < nums.size()-1) {
            int far = 0;
            for (int i=m+1; i<=M; i++) {
                tmp = i + nums[i];
                if (tmp > far) { far = tmp; }
            }
            m = M;
            M = far;
            cnt++;
        }
        return cnt;
    }
};
